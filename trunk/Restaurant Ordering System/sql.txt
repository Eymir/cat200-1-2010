CREATE TABLE FOODTABLE (
FOOD_NO NUMBER(6) PRIMARY KEY,
FOOD_NAME VARCHAR2(50) NOT NULL,
FOOD_TYPE VARCHAR2(30) NOT NULL,
FOOD_PRICE NUMBER(10,2) NOT NULL,
FOOD_QOH NUMBER(10) NOT NULL,
FOOD_MIN NUMBER(10) NOT NULL,
picture  VARCHAR2(20) NOT NULL
);

CREATE TABLE TABLEINFO(
TABLE_NO  NUMBER(5)   NOT NULL PRIMARY KEY,
TABLE_STATUS    VARCHAR2(5)  NOT NULL  
);

CREATE TABLE RECEIPTTABLE (
RECEIPT_NO NUMBER(10) PRIMARY KEY,
RECEIPT_DATE DATE DEFAULT (sysdate),
TABLE_NO NUMBER(5) NOT NULL,
PRICE_CHARGED NUMBER(10,2) NOT NULL,
FOREIGN KEY(TABLE_NO) REFERENCES TABLEINFO(TABLE_NO)
);

CREATE TABLE ORDERTABLE (
ORDER_NO NUMBER(10) PRIMARY KEY,
RECEIPT_NO NUMBER(10) NOT NULL,
ORD_FOOD_NO NUMBER(6) NOT NULL,
ORD_FOOD_QN NUMBER(10) NOT NULL,
FOREIGN KEY (RECEIPT_NO) REFERENCES RECEIPTTABLE(RECEIPT_NO),
FOREIGN KEY (ORD_FOOD_NO) REFERENCES FOODTABLE(FOOD_NO)
);

CREATE TABLE ADMINTABLE (
USER_ID VARCHAR2(30) PRIMARY KEY,
USER_PASSWORD VARCHAR(30) NOT NULL
);
